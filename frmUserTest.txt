Private Sub custom_KeyDown(ByVal KeyCode As MSForms.ReturnInteger, ByVal Shift As Integer)
  Dim Icount As Integer
  Dim TooHigh As Integer
  Const TooLow = 0
  Const No_Selection = -1
  Const EnterKey = 13
  
  Select Case KeyCode
     Case EnterKey
       Label1.Caption = "Pressed ENTER"
     Case vbKeyF2
       Label1.Caption = "F2 pressed"
     Case vbKeyUp
       Label1.Caption = "Up arrow"
       Icount = ListBox1.ListIndex
       TooHigh = ListBox1.ListCount - 1
       
       'MsgBox str(icount)
        Select Case Icount
          Case No_Selection
            'Select first item
            ListBox1.ListIndex = 0
          Case TooLow
            'Wrap back to first item
            ListBox1.ListIndex = ListBox1.ListCount - 1
          Case Else
            ListBox1.ListIndex = Icount - 1
        End Select
    Case vbKeyDown
     Label1.Caption = "down arrow"
     Icount = ListBox1.ListIndex
     TooHigh = ListBox1.ListCount - 1
     'MsgBox str(icount)
        Select Case Icount
          Case No_Selection
            'Select first item
            ListBox1.ListIndex = 0
          Case TooHigh
            'Wrap back to first item
            ListBox1.ListIndex = 0
          Case Else
            ListBox1.ListIndex = Icount + 1
        End Select
   Case Else
    'do nothing
  End Select
End Sub
